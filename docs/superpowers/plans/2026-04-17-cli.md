# highstack-ui CLI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a zero-dependency Node.js CLI that copies highstack-ui components into Angular projects via `npx highstack-ui init` and `npx highstack-ui add <component>`.

**Architecture:** A `cli/` directory at the repo root containing ES modules (.mjs) with no external dependencies. Component files are embedded as templates and copied as-is. A registry object maps component names to their metadata and files.

**Tech Stack:** Node.js (fs, path, process), ES modules

---

## File Structure

```
cli/
  package.json                              # npm package config with bin entry
  bin/
    index.mjs                               # entry point, command router
  src/

    components/
      button/
        button.component.ts                 # copy of src/components/atoms/button/button.component.ts

```

```typescript
export type ButtonVariant = 'default' | 'destructive' | 'outline' | 'ghost' | 'link' | 'secondary';
export type ButtonSize = 'sm' | 'md' | 'lg' | 'icon';

@Component({
  selector: 'ui-button',
  templateUrl: './button.component.html',
})
export class ButtonComponent {
  readonly variant = input<ButtonVariant>('default');
  readonly size = input<ButtonSize>('md');
  readonly disabled = input<boolean>(false);
  readonly loading = input<boolean>(false);
  readonly type = input<'button' | 'submit' | 'reset'>('button');

  protected readonly isDisabled = computed(() => this.disabled() || this.loading());

  protected readonly hostClasses = computed(() => {
    const base = [
      'inline-flex items-center justify-center gap-2 whitespace-nowrap rounded-[var(--radius)]',
      'text-sm font-medium transition-all',
      'outline-none focus-visible:ring-[3px] focus-visible:ring-[var(--color-ring)]/50',
      'disabled:pointer-events-none disabled:opacity-50',
      'cursor-pointer',
    ].join(' ');

    const variantMap: Record<ButtonVariant, string> = {
      default:
        'bg-[var(--color-primary)] text-[var(--color-primary-foreground)] shadow-xs hover:bg-[var(--color-primary)]/90',
      destructive:
        'bg-[var(--color-destructive)] text-[var(--color-destructive-foreground)] shadow-xs hover:bg-[var(--color-destructive)]/90 focus-visible:ring-[var(--color-destructive)]/20',
      outline:

      secondary:

      ghost:

      link: 'text-[var(--color-primary)] underline-offset-4 hover:underline',
    };

      md: 'h-9 px-4 py-2',
      lg: 'h-10 px-6',
```

```javascript
import { existsSync, mkdirSync, copyFileSync } from 'node:fs';
import { dirname } from 'node:path';

const green = (s) => `\x1b[32m${s}\x1b[0m`;
const red = (s) => `\x1b[31m${s}\x1b[0m`;
const cyan = (s) => `\x1b[36m${s}\x1b[0m`;
const dim = (s) => `\x1b[2m${s}\x1b[0m`;

export const log = {
  success: (msg) => console.log(`${green('✓')} ${msg}`),
  error: (msg) => console.error(`${red('✗')} ${msg}`),
  skip: (msg) => console.log(`${dim('⊘')} ${msg}`),
  info: (msg) => console.log(`${cyan('ℹ')} ${msg}`),
};

export function copyTemplate(src, dest) {
  if (existsSync(dest)) {
    log.skip(`Skipped ${dest} (already exists)`);
    return false;
  }
  mkdirSync(dirname(dest), { recursive: true });
  copyFileSync(src, dest);
  log.success(`Created ${dest}`);
  return true;
}
```

- [ ] **Step 1: Create init.mjs**

```javascript
import { existsSync, readFileSync, writeFileSync } from 'node:fs';
import { resolve, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';
import { log, copyTemplate } from './utils.mjs';

const __dirname = dirname(fileURLToPath(import.meta.url));
const templatesDir = resolve(__dirname, '..', 'templates');

export function init() {
  const cwd = process.cwd();

  if (!existsSync(resolve(cwd, 'angular.json'))) {
    log.error('Not an Angular project. Run this command from the root of an Angular project.');
    process.exit(1);
  }

  // Copy tokens.css
  copyTemplate(
    resolve(templatesDir, 'tokens.css'),
    resolve(cwd, 'src', 'styles', 'tokens.css'),
  );

  // Update styles.css
  const stylesPath = resolve(cwd, 'src', 'styles.css');
  const importLine = '@import "./styles/tokens.css";';

  if (existsSync(stylesPath)) {
    const content = readFileSync(stylesPath, 'utf-8');
    if (!content.includes(importLine)) {
      writeFileSync(stylesPath, `${importLine}\n${content}`);
      log.success('Updated src/styles.css');
    } else {
      log.skip('src/styles.css already imports tokens');
    }
  } else {
    writeFileSync(stylesPath, `${importLine}\n`);
    log.success('Created src/styles.css');
  }

  console.log('\nReady! Run: npx highstack-ui add button');
}
```

```javascript
import { existsSync } from 'node:fs';
import { resolve, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';
import { log, copyTemplate } from './utils.mjs';
import { registry } from './registry.mjs';

const __dirname = dirname(fileURLToPath(import.meta.url));
const templatesDir = resolve(__dirname, '..', 'templates', 'components');

export function add(components) {
  const cwd = process.cwd();

  if (!existsSync(resolve(cwd, 'src', 'styles', 'tokens.css'))) {
    log.error('highstack-ui not initialized. Run: npx highstack-ui init');
    process.exit(1);
  }

  if (components.length === 0) {
    log.error('No component specified. Usage: npx highstack-ui add <component>');
    process.exit(1);
  }

  for (const name of components) {
    const entry = registry[name];
    if (!entry) {
      log.error(`Unknown component '${name}'. Run: npx highstack-ui list`);
      continue;
    }

    for (const file of entry.files) {
      const src = resolve(templatesDir, entry.name, file);
      const dest = resolve(cwd, 'src', 'components', entry.category, entry.name, file);
      copyTemplate(src, dest);
    }

    console.log(`\nImport it:\n  import { ButtonComponent } from './components/${entry.category}/${entry.name}/${entry.files[0].replace('.ts', '')}';`);
  }
}
```

### Task 7: Entry point

**Files:**

- [ ] **Step 1: Create bin/index.mjs**

```javascript
#!/usr/bin/env node

import { init } from '../src/init.mjs';
import { add } from '../src/add.mjs';
import { registry } from '../src/registry.mjs';

const [,, command, ...args] = process.argv;

switch (command) {
  case 'init':
    init();
    break;

  case 'add':
    add(args);
    break;

  case 'list': {
    console.log('\nAvailable components:');
    for (const [name, entry] of Object.entries(registry)) {
      console.log(`  ${name.padEnd(12)} ${entry.description}`);
    }
    console.log('');
    break;
  }

  default:
    console.log(`
highstack-ui — Copy-paste Angular component library

Usage:
  npx highstack-ui init              Set up tokens and styles
  npx highstack-ui add <component>   Copy a component into your project
  npx highstack-ui list              List available components
`);
    break;
}
```
